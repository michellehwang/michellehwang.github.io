    '''
    >>> orig_tup = (2,4,6,8)
    >>> add_four = lambda x : x + 4
    >>> tuple_replace(orig_tup, 2)(add_four)
    (2, 4, 10, 12)
    >>> tuple_replace(orig_tup, 5)(lambda x : x * x)
    (2, 4, 6, 8)
    '''

def tuple_replace(tup, n):
    def gen(fn):
        count, result, apply = 0, (), False
        while count < len(tup):
            if count == n:
                apply = True
            if apply:
                result += (fn(tup[count]),)
            else:
                result += (tup[count],)
            count += 1
        return result
    return gen

Implement a function uniq, which takes in a sorted list of numbers that might contain duplicates. Mutate the list such that it only contains one instance of each number (i.e. remove the duplicates). Do not return or create any new lists in your implementation (remember, slicing a list will create a new list, so don't do it!). 
def uniq(lst):
     """Takes a sorted list of numbers and mutatively removes all duplicates
     from the list.
     >>> lst = [1, 1, 2, 4, 4, 9, 10, 10, 10, 10]
     >>> uniq(lst) # doesn't return anything!
     >>> lst
     [1, 2, 4, 9, 10]
     >>> empty = []
     >>> uniq(empty)
     >>> empty
     []
     """

def uniq(lst):
   if len(lst) < 2:
       return
   i = len(lst) - 2
   while i >= 0:
       if lst[i] == lst[i + 1]:
           lst.pop(i + 1)
       i -= 1

For these questions, assume you have the following RList implementation: 
class RList:
    class EmptyList:
        def __len__(self):
            return 0

    empty = EmptyList()

    def __init__(self, first, rest=empty):
        self.first = first
        self.rest = rest

    def __repr__(s):
        """Return a string that would evaluate to s."""
        if s.rest is RList.empty:
            rest = ''
        else:
            rest = ', ' + repr(s.rest)
        return 'RList({0}{1})'.format(s.first, rest)
Implement a function combine_adjacent, which takes in an RList and condenses adjacent, repeated RList entries. First, try writing this such that you create and return a new RList. 
def compress_adjacent(rlst):
    """
    >>> all_same = RList(1, RList(1, RList(1)))
    >>> combine_adjacent(all_same)
    RList(1)
    >>> long = RList(1, RList(1, RList(2, RList(4, RList(4, RList(9, RList(10, RList(10, RList(10, RList(10))))))))))
    >>> combine_adjacent(long)
    RList(1, RList(2, RList(4, RList(9, RList(10)))))
    """


def combine_adjacent(rlst):
    new = RList(rlst.first)
    new_ptr = new # Keep a pointer to the rlist so we can return it at the end!
    curr = rlst.first
    rlst = rlst.rest
    while rlst != RList.empty:
        if rlst.first != curr:
            new.rest = RList(rlst.first)
            new = new.rest
            curr = rlst.first
        rlst = rlst.rest
    return new_ptr

def compress_adjacent_mutate(rlst):
    """
    >>> all_same = RList(1, RList(1, RList(1)))
    >>> combine_adjacent_mutate(all_same)
    >>> all_same
    RList(1)
    >>> long = RList(1, RList(1, RList(2, RList(4, RList(4, RList(9, RList(10, RList(10, RList(10, RList(10))))))))))
    >>> combine_adjacent_mutate(long)
    >>> long
    RList(1, RList(2, RList(4, RList(9, RList(10)))))
    """


def compress_adjacent_mutate(rlst):
    curr = rlst.first
    while rlst.rest is not RList.empty:
        if rlst.rest.first == curr:
            rlst.rest = rlst.rest.rest
        else:
            curr = rlst.rest.first
            rlst = rlst.rest

Write a function merge_unique_values which takes in two dictionaries first and second that merges the values of second into first if and only if they are not already in first. Keys that exist in second, but do not exist must be added to first 

def merge_unique_values(first, second):
    """
    >>> first = {'a': [1, 1, 2, 3], 'r': [1, 2, 5, 6]}
    >>> second = {'a': [1, 1, 2, 3, 4], 'r': [1, 2, 5, 6], 'm': [15, 6, 7]}
    >>> merge_unique_values(first, second)
    >>> first
    {'a': [1, 1, 2, 3, 4], 'r': [1, 2, 5, 6], 'm': [15, 6, 7]}
    >>> second
    {'a': [1, 1, 2, 3, 4], 'r': [1, 2, 5, 6], 'm': [15, 6, 7]}
    >>> first = {'a': ['herp', 'derp'], 'j': ['foo', 'bar'], 'c': []}
    >>> second = {'c': ['hello', 'world']}
    >>> merge_unique_values(first, second)
    >>> first
    {'a': ['herp', 'derp'], 'c': ['hello', 'world'], 'j': ['foo', 'bar']}
    """


def merge_unique_values(first, second):
    for key, val in second.items():
        if key in first:
            for v in val:
                if v not in first[key]:
                    first[key].append(v)
        else:
            first[key] = second[key]
