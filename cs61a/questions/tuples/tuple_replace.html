def tuple_replace(tup, n):
    def gen(fn):
        count, result, apply = 0, (), False
        while count < len(tup):
            if count == n:
                apply = True
            if apply:
                result += (fn(tup[count]),)
            else:
                result += (tup[count],)
            count += 1
        return result
    return gen
